                        // Structure for Heap Tree
                        struct Heap {
                            int* array;
                            int capacity;
                            int size;
                        };
                        
                        // Function to create a new heap tree
                        struct Heap* createHeap(int capacity) {
                            struct Heap* heap = (struct Heap*)malloc(sizeof(struct Heap));
                            heap->array = (int*)malloc(capacity * sizeof(int));
                            heap->capacity = capacity;
                            heap->size = 0;
                            return heap;
                        }
                        
                        // Function to swap two elements in an array
                        void swap(int* a, int* b) {
                            int temp = *a;
                            *a = *b;
                            *b = temp;
                        }
                        
                        // Function to maintain the heap property (upward)
                        void heapifyUp(struct Heap* heap, int index) {
                            int parent = (index - 1) / 2;
                        
                            if (parent >= 0 && heap->array[parent] > heap->array[index]) {
                                swap(&heap->array[parent], &heap->array[index]);
                                heapifyUp(heap, parent);
                            }
                        }
                        
                        // Function to maintain the heap property (downward)
                        void heapifyDown(struct Heap* heap, int index) {
                            int smallest = index;
                            int left = 2 * index + 1;
                            int right = 2 * index + 2;
                        
                            if (left < heap->size && heap->array[left] < heap->array[smallest])
                                smallest = left;
                        
                            if (right < heap->size && heap->array[right] < heap->array[smallest])
                                smallest = right;
                        
                            if (smallest != index) {
                                swap(&heap->array[index], &heap->array[smallest]);
                                heapifyDown(heap, smallest);
                            }
                        }
                        
                        // Function to insert an element into the heap tree
                        void insert(struct Heap* heap, int value) {
                            if (heap->size == heap->capacity) {
                                printf("Heap Overflow\n");
                                return;
                            }
                        
                            heap->array[heap->size] = value;
                            heapifyUp(heap, heap->size);
                            heap->size++;
                        }
                        
                        // Function to extract the minimum value from the heap tree
                        int extractMin(struct Heap* heap) {
                            if (heap->size == 0) {
                                printf("Heap Underflow\n");
                                return -1;
                            }
                        
                            int min = heap->array[0];
                            heap->array[0] = heap->array[heap->size - 1];
                            heap->size--;
                            heapifyDown(heap, 0);
                            return min;
                        }

                                printf("%d ", heap->array[i]);

                            free(heap);
